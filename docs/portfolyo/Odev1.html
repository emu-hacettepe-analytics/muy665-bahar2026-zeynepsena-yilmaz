  1. MÜY665 Ödev-1: Veri Analizi ve Temizleme
  1. ADIM: Paketlerin Yüklenmesi

{r} #| message: false library(dslabs) library(dplyr) library(tidyr)

  1. ADIM: Kişisel Bilgilerin Tanımlanması

{r}

my_first <- “Zeynep Sena”
my_birth_year <- 1997

  1. ADIM: k Değerinin Hesaplanması

{r}

k <- (nchar(my_first) + my_birth_year) %% 15 + 8 print(paste(“Hesaplanan k degeri:”, k))

  1. ADIM: Verilerin Çağrılması

{r}

data(“polls_us_election_2016”) original_data <- polls_us_election_2016

k kuralına göre görüntüleme (k çift ise başı, tek ise sonu)

if (k %% 2 == 0) { head(original_data, k) } else { tail(original_data, k) }

  1. ADIM: Hataların Düzeltilmesi

{r} # Toplam NA sayısı print(paste(“Toplam NA sayısı:”, sum(is.na(original_data))))

Sütun başına NA sayıları (İlk 8 sütun)

na_counts <- sort(colSums(is.na(original_data)), decreasing = TRUE) head(na_counts, 8)

new_data <- original_data %>%

mutate(across(where(is.numeric), ~replace_na(.x, my_birth_year + k))) %>% mutate(across(where(~is.character(.x) | is.factor(.x)), ~replace_na(as.character(.x), paste0(my_first, “_“, k))))

  1. ADIM: Kontrol ve Doğrulama

{r}

Yeni veri çerçevesi için k kuralına göre görüntüleme

if (k %% 2 == 0) { head(new_data, k) } else { tail(new_data, k) }

Toplam NA sayısı 0 çıkmalı

print(paste(“Yeni veri setindeki toplam NA:”, sum(is.na(new_data))))

Doğrulama (Sonuç FALSE çıkmalı)

print(anyNA(new_data))
